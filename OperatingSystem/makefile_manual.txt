makefile文件一般格式：
target：prerequisite
    command

cc = gcc				//相当于C语言中的宏
prom = calc
deps = $(shell find ./ -name "*.h") 	//使用shell命令找到当前目录下.h依赖文件
src = $(shell find ./ -name "*.c")	//使用shell命令找到当前目录下.c原文件
obj = $(src:%.c=%.o) 
    
$(prom): $(obj)
       $(cc) -o $(prom) $(obj)

%.o: %.c $(deps)
        $(cc) -c $< -o $@
clean:
        rm -rf $(obj) $(prom)
